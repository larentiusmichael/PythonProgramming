main_menu_function
Begin
	Declare opt
	Display "\t\t*****WELCOME TO REAL CHAMPIONS SPORT  ACADEMY*****\nSelect User Type:\n1. Admin\n2. Registered Student\n3. Non Registered Student"
	Read opt
	While (opt > 3 OR opt < 1)
		Display "Wrong Choice!\nKindly select user type again:\n1. Admin\n2. Registered Student\n3. Non Registered Student"
		Read opt
	End while
	if (opt == 1)
		call admin_function
	else if (opt == 2)
		call student_function
	else
		call nonstudent_function
	End if
End

coach_modify_admin_function

	Read name_coach
	no = 0
	words = []
	Open File "coach.txt" For Read Mode
		data = file.readlines()
	Close File
	Open File "coach.txt" For Read Mode
	LOOP line in File
		Append line in words
	End LOOP
	Close File

		else

		End if

modify_student_function(username_murid)
Begin
	Declare nama_pengguna, no, words, data, line, cnt, hao
	nama_pengguna = username_murid
	no = 0
	words = []
	Open File "student.txt" For Read Mode
		data = file.readlines()
	Close File
	Open File "student.txt" For Read Mode
	LOOP line in File
		Append line in words
	End LOOP
	Close File
	LOOP cnt in words
		if nama_pengguna in cnt
			hao = no
			call modify_student_second_function(hao,data)
		else
			no = no + 1
		End if
	End LOOP
End

Begin
	Declare nonstudent_option
	Display "\t\t*****REAL CHAMPIONS SPORT  ACADEMY*****\nSelect one specific function:\n1. View Detail of Sport\n2. View Detail of Sport Schedule\n3. Register\n4. Exit"
	Read nonstudent_option
	While (nonstudent_option > 4 or nonstudent_option < 1)
		Display "Wrong Choice!\nSelect one specific function:\n1. View Detail of Sport\n2. View Detail of Sport Schedule\n3. Register\n4. Exit"
		Read nonstudent_option
	End while
	if (nonstudent_option == 1)
		call sport_nonstudent_function
	else if (nonstudent_option == 2)
		call schedule_nonstudent_function
	else if (nonstudent_option == 3)
		call register_nonstudent_function
	else
		call main_menu_function
	End if
End
